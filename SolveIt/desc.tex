\problemtitle{Solve It!}

\begin{problem}

Solve the equation:
\begin{center}
$p*e^{-x} + q*\sin(x) + r*\cos(x) + s*\tan(x) + t * x^2 + u = 0$ 
\end{center}
where $0 \le x \le 1$.
\end{problem}

\begin{inputdescription}
The input consists of multiple test cases and is terminated by \code{EOF}. Each test case consists of 6 integers in a single line: $p,q,r,s,t$ and $u$, where $0 \le p,r \le 20$ and $-20 \le q,s,t \le 0$ and $u \in \mathbb{Z}$ can stored in a standard 32-bit integer. There will be a maximum 2100 lines in the input file.
\end{inputdescription}

\begin{outputdescription}
For each set of input, print a line containing the value of $x$, rounded to 4 decimal places, or the string \code{'No solution'}, whichever is applicable.
\end{outputdescription}
